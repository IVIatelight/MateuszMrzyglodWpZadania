CREATE TABLE Users(  
   ID INT IDENTITY NOT NULL PRIMARY KEY,   
   Name VARCHAR(40) NOT NULL,
   DESCRIPTION VARCHAR(200)
);

CREATE TABLE Products(  
   ID INT IDENTITY NOT NULL PRIMARY KEY,   
   Name VARCHAR(40) NOT NULL,
   DESCRIPTION VARCHAR(200)
);

CREATE TABLE Privileges(  
   ID INT IDENTITY NOT NULL PRIMARY KEY,   
   Name VARCHAR(40) NOT NULL,
   DESCRIPTION VARCHAR(200)
);
