CREATE TABLE PRIVILEGES(
ID_PRIVILEGES INTEGER CONSTRAINT PK_PRIVILEGES PRIMARY KEY,
NAME_PRIVILEGES VARCHAR(30) UNIQUE NOT NULL,
DESCRIPTION_PRIVILEGES VARCHAR(200),
);

CREATE TABLE USERS(
ID_USER INTEGER PRIMARY KEY,
FIRST_NAME VARCHAR(30),
LAST_NAME VARCHAR(30),
PRIVILEGES_ID INTEGER NOT NULL CONSTRAINT FK_ID_PRIVILEGES REFERENCES PRIVILEGES(ID_PRIVILEGES)
);
